### Task 5: Fix lib/api.ts

**Files:**
- Modify: `apps/web/lib/api.ts`

**Context:** The existing `apps/web/lib/api.ts` uses a nested namespace shape (`api.cases.list()`) and has a wrong `ReviewPayload` type (`{ answers: Record<string, string> }`). The spec requires top-level named exports: `getCases()`, `getCase(id)`, `submitReview(payload)`, `getScore(reviewId)`. Types come from `@ledgerlens/types`. The POST `/reviews` body is `{ case_id, action, reasoning, time_spent_seconds }` and it returns `{ review_id: string }`. GET `/reviews/{id}/score` returns either a full `Score` or `{ status: 'pending' }` while the background task is still running.

- [ ] **Step 1: Rewrite lib/api.ts**

Replace `apps/web/lib/api.ts`:
```ts
import type { Case, Review, Score } from '@ledgerlens/types'

const BASE = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000'

async function apiFetch<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${BASE}${path}`, {
    headers: { 'Content-Type': 'application/json' },
    ...init,
  })
  if (!res.ok) throw new Error(`API ${res.status}: ${path}`)
  return res.json() as Promise<T>
}

// Public case shape — API strips hiddenTruth before sending
export type CasePublic = Omit<Case, 'hiddenTruth'>

export type ReviewCreate = {
  case_id: string
  action: Review['action']
  reasoning: string
  time_spent_seconds: number
}

export type ReviewCreated = { review_id: string }

export type ScoreOrPending = Score | { status: 'pending' }

export function getCases(): Promise<CasePublic[]> {
  return apiFetch<CasePublic[]>('/cases')
}

export function getCase(id: string): Promise<CasePublic> {
  return apiFetch<CasePublic>(`/cases/${id}`)
}

export function submitReview(payload: ReviewCreate): Promise<ReviewCreated> {
  return apiFetch<ReviewCreated>('/reviews', {
    method: 'POST',
    body: JSON.stringify(payload),
  })
}

export function getScore(reviewId: string): Promise<ScoreOrPending> {
  return apiFetch<ScoreOrPending>(`/reviews/${reviewId}/score`)
}
```

### Task 6: TanStack Query provider + hooks

**Files:**
- Create: `apps/web/app/providers.tsx`
- Create: `apps/web/lib/hooks.ts`

**Context:** TanStack Query v5 requires a `QueryClient` instance and a `QueryClientProvider` wrapping the tree. It must be a Client Component because it uses `useState`. The hooks file provides four hooks that components will use instead of calling `api.*` directly. `useScore` polls at 2-second intervals while the score is pending (the API returns `{ status: 'pending' }` until the background scoring task completes).

- [ ] **Step 1: Create app/providers.tsx**

Create `apps/web/app/providers.tsx`:
```tsx
'use client'

import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { useState } from 'react'

export function Providers({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 60_000,
            retry: 1,
          },
        },
      })
  )

  return <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
}
```

- [ ] **Step 2: Create lib/hooks.ts**

Create `apps/web/lib/hooks.ts`:
```ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { getCases, getCase, submitReview, getScore } from '@/lib/api'
import type { ReviewCreate } from '@/lib/api'

export function useCases() {
  return useQuery({
    queryKey: ['cases'],
    queryFn: getCases,
  })
}

export function useCase(id: string) {
  return useQuery({
    queryKey: ['case', id],
    queryFn: () => getCase(id),
    enabled: Boolean(id),
  })
}

export function useSubmitReview() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (payload: ReviewCreate) => submitReview(payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['cases'] })
    },
  })
}

export function useScore(reviewId: string | null) {
  return useQuery({
    queryKey: ['score', reviewId],
    queryFn: () => getScore(reviewId!),
    enabled: Boolean(reviewId),
    refetchInterval: (query) => {
      const data = query.state.data
      if (!data || 'status' in data) return 2_000
      return false
    },
  })
}
```
